        <div style="display: flex; gap: 10px;">

                <option value="click">Check</option>

        <button class="btn btn-add" onclick="addTask()">+ A√ëADIR A MI RUTINA</button>

    <div class="task-list" id="taskList"></div>

    <script>
        let totalSeconds = parseInt(localStorage.getItem('totalSeconds')) || 0;
        let streak = parseInt(localStorage.getItem('streak')) || 0;
        let lastDate = localStorage.getItem('lastDate') || "";
        let isUsing = false;
        let mainInterval;

        function updateDisplay() {
            let h = Math.floor(totalSeconds / 3600);
            let m = Math.floor((totalSeconds % 3600) / 60);
            let s = totalSeconds % 60;
            document.getElementById('balance').innerText = 
                `${h.toString().padStart(2,'0')}:${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
            localStorage.setItem('totalSeconds', totalSeconds);
            document.getElementById('streakDisplay').innerText = `üî• ${streak} D√çAS`;
        }

        function addTask() {
            const name = document.getElementById('taskName').value;
            const reward = document.getElementById('taskReward').value;
            const type = document.getElementById('taskType').value;
            if (!name || !reward) return;

            const id = Date.now();
            const taskHtml = `
                <div class="task-item" id="task-${id}">
                    <div class="task-info">
                        <b>${name}</b>
                        <span id="label-${id}">Recompensa: +${reward}m</span>
                    </div>
                    <button class="btn btn-action" id="btn-${id}" onclick="${type === 'click' ? `claim(${reward*60}, '${id}')` : `startTaskTimer(${reward*60}, '${id}')`}">
                        ${type === 'click' ? 'Hecho' : 'Iniciar'}
                    </button>
                </div>`;
            
            document.getElementById('taskList').insertAdjacentHTML('afterbegin', taskHtml);
            saveTasks();
            document.getElementById('taskName').value = "";
            document.getElementById('taskReward').value = "";
        }

        function claim(seconds, id) {
            totalSeconds += seconds;
            updateStreak();
            updateDisplay();
            if(id) {
                const btn = document.getElementById(`btn-${id}`);
                btn.innerText = "¬°+"+(seconds/60)+"m!";
                setTimeout(() => btn.innerText = "Hecho", 2000);
            }
        }

        function updateStreak() {
            const today = new Date().toDateString();
            if (lastDate !== today) {
                streak++;
                lastDate = today;
                localStorage.setItem('streak', streak);
                localStorage.setItem('lastDate', lastDate);
            }
        }

        function startTaskTimer(seconds, id) {
            let remaining = seconds;
            const btn = document.getElementById(`btn-${id}`);
            const label = document.getElementById(`label-${id}`);
            btn.disabled = true;
            btn.style.opacity = "0.5";
            
            const t = setInterval(() => {
                remaining--;
                let m = Math.floor(remaining/60);
                let s = remaining%60;
                label.innerText = `‚è≥ Quedan: ${m}:${s.toString().padStart(2,'0')}`;
                label.classList.add("timer-running");
                
                if (remaining <= 0) {
                    clearInterval(t);
                    claim(seconds, id);
                    label.innerText = "¬°Completado!";
                    label.classList.remove("timer-running");
                    btn.disabled = false;
                    btn.style.opacity = "1";
                    btn.innerText = "Reiniciar";
                }
            }, 1000);
        }

        function toggleUsage() {
            const btn = document.getElementById('useBtn');
            if (!isUsing) {
                if (totalSeconds <= 0) return alert("Sin tiempo disponible.");
                isUsing = true;
                btn.innerText = "PAUSAR (GUARDAR)";
                btn.classList.add("btn-pause");
                mainInterval = setInterval(() => {
                    if (totalSeconds > 0) {
                        totalSeconds--;
                        updateDisplay();
                    } else { stopUsage(); }
                }, 1000);
            } else { stopUsage(); }
        }

        function stopUsage() {
            isUsing = false;
            clearInterval(mainInterval);
            const btn = document.getElementById('useBtn');
            btn.innerText = "CANJEAR TIEMPO";
            btn.classList.remove("btn-pause");
        }

        function saveTasks() {
            localStorage.setItem('savedTasksHtml', document.getElementById('taskList').innerHTML);
        }

        window.onload = () => {
            const h = localStorage.getItem('savedTasksHtml');
            if (h) document.getElementById('taskList').innerHTML = h;
            updateDisplay();
        };
    </script>
